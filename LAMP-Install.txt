# Install Java 
- sudo apt-get update
- sudo apt-get upgrade
- sudo apt-get install default-jdk -y
- sudo apt-get install ant git vim -y

# Install LAMP
- sudo apt-get install apache2 -y
- sudo vim /etc/apache2/apache2.conf
# Bottom of the file append - ServerName 127.0.0.1
# Test apache
- sudo apache2ctl configtest
# There should be : Syntax OK
- sudo apt-get install php libapache2-mod-php php-mcrypt php-mysql -y
- sudo /etc/init.d/apache2 restart
- sudo apt-get install mysql-server
- sudo mysql_secure_installation
# Select y, 0, enter password "pass", n, n, n, y
- sudo mysql -u root -p
- CREATE DATABASE data;
- show databases;
- GRANT ALL PRIVILEGES ON data.* TO 'root'@'localhost' IDENTIFIED BY 'pass';
- FLUSH PRIVILEGES;
- exit;
- sudo apt-get install phpmyadmin
# Choose apache2
# Configure phpmyadmin : Yes
# Enter MySQL password - "pass"
# Enter user password
- sudo vim /etc/php5/apache2/php.ini
# Uncomment the line - extension=mysql.so
- sudo vim /etc/apache2/apache2.conf
# In the end write - Include /etc/phpmyadmin/apache.conf
# Restart apache service
- sudo /etc/init.d/apache2 restart
